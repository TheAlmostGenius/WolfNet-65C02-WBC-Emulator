402  private void BackgroundWorkerDoWork(object sender, DoWorkEventArgs e)
403  {
404  var worker = sender as BackgroundWorker;
405  var outputLogs = new List<OutputLog>();
406 
407  while (true)
408  {
409  if (worker != null && worker.CancellationPending || IsBreakPointTriggered())
410  {
411  e.Cancel = true;
412 
413  RaisePropertyChanged("Proc");
414 
415  foreach (var log in outputLogs)
416  OutputLog.Insert(0, log);
417 
418  UpdateMemoryPage();
419  return;
420  }
421 
422  StepProcessor();
423  outputLogs.Add(GetOutputLog());
424 
425  if (NumberOfCycles % GetLogModValue() == 0)
426  {
427  foreach (var log in outputLogs)
428  OutputLog.Insert(0, log);
429

431  UpdateUi();
432  }

434  }

437  private bool IsBreakPointTriggered()
438  {
439  //This prevents the Run Command from getting stuck after reaching a breakpoint
440  if (_breakpointTriggered)
441  {
442  _breakpointTriggered = false;
443  return false;
444  }
445 
446  foreach (var breakpoint in Breakpoints.Where(x => x.IsEnabled))
447  {
448  int value;
449 
450  if (!int.TryParse(breakpoint.Value, NumberStyles.AllowHexSpecifier, CultureInfo.InvariantCulture, out value))
451  continue;
452 
453  if (breakpoint.Type == BreakpointType.NumberOfCycleType && value == NumberOfCycles)
454  {
455  _breakpointTriggered = true;
456  RunPause();
457  return true;
458  }
459 
460  if (breakpoint.Type == BreakpointType.ProgramCounterType && value == W65C02.ProgramCounter)
461  {
462  _breakpointTriggered = true;
463  RunPause();
464  return true;
465  }
466  }
467 
468  return false;
469  }
